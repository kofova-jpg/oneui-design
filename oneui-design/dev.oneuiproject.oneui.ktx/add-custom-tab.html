inline fun <T : TabLayout> T.addCustomTab(@StringRes tabTitleRes: Int?, @DrawableRes tabIconRes: Int?, listener: View.OnClickListener): TabLayout.Tab(source)

Adds a custom tab to this TabLayout. A custom tab has no selected state.

Return

The added TabLayout.Tab instance for further configuration, if needed.

Example usage:

val tab = tabLayout.addCustomTab(
            tabTitleRes = R.string.tab_title,
            tabIconRes = R.drawable.tab_icon) {view ->
              // Handle tab click
           }
